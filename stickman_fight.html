    class BloodParticle {
        constructor(x, y, color) {
            this.x = x;
            this.y = y;
            this.vx = (Math.random() - 0.5) * 8;
            this.vy = (Math.random() - 0.5) * 8 - 2;
            this.size = Math.random() * 4 + 2;
            this.life = 1.0;
            this.gravity = 0.3;
            this.color = color; // Spielerfarbe speichern
        }
        
        update() {
            this.x += this.vx;
            this.y += this.vy;
            this.vy += this.gravity;
            this.life -= 0.02;
            return this.life > 0;
        }
        
        draw() {
            // RGB-Werte aus der Spielerfarbe extrahieren
            let r, g, b;
            if (this.color === '#00d2ff') {
                // Blau (Spieler 1)
                r = 0;
                g = 210;
                b = 255;
            } else {
                // Rot (Spieler 2)
                r = 255;
                g = 65;
                b = 108;
            }
            
            ctx.fillStyle = `rgba(${r}, ${g}, ${b}, ${this.life})`;
            ctx.beginPath();
            ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
            ctx.fill();
        }
    }

    function updateBot() {
        if (gameMode !== 'bot') return;

        const verticalDist = Math.abs(player2.y - player1.y);

        if (botTimer % 20 === 0) {
            if (distance > 100) {
                // N√§her kommen

            if (action.type === 'move') {

    function checkHit(attacker, defender) {
        const distance = Math.abs(attacker.x - defender.x);
        const verticalDist = Math.abs(attacker.y - defender.y);
        
        if (distance < 80 && verticalDist < 50) {
            defender.health -= 5;
            updateHealthBar(defender === player1 ? 1 : 2);
            
            // Blut-Effekt in der Farbe des getroffenen Spielers
            createBloodEffect(defender.x, defender.y - 20, defender.color);
            
            // R√ºcksto√ü

                defender.velocityX = 8;
            } else {
                defender.velocityX = -8;
            }
            
            if (defender.health <= 0) {
                gameOver(attacker === player1 ? 1 : 2);
            }
        }
    }

    function updatePlayer(player, leftKey, rightKey, upKey, touchPlayer) {
        // Bewegung (Tastatur oder Touch)
        const leftPressed = keys[leftKey] || (touchPlayer && touchControls[touchPlayer].left);
        const rightPressed = keys[rightKey] || (touchPlayer && touchControls[touchPlayer].right);
        const upPressed = keys[upKey] || (touchPlayer && touchControls[touchPlayer].up);
        
        if (leftPressed) {
            player.velocityX = -moveSpeed;
            player.facingRight = false;
        } else if (rightPressed) {
            player.velocityX = moveSpeed;
            player.facingRight = true;
        } else {
            player.velocityX *= 0.8;
        }
        
        // Springen
        if (upPressed && player.y >= groundY) {
            player.velocityY = -jumpPower;
        }
        
        // Gravitation
        player.velocityY += gravity;
        
        // Position aktualisieren
        player.x += player.velocityX;
        player.y += player.velocityY;
        
        // Boden
        if (player.y > groundY) {
            player.y = groundY;
            player.velocityY = 0;
        }
        
        // Grenzen
        player.x = Math.max(30, Math.min(770, player.x));
        
        // Angriff-Cooldown
        if (player.attackCooldown > 0) {
            player.attackCooldown--;
        }
        if (player.attackCooldown === 0) {
            player.attacking = false;
        }
        
        // Special-Cooldown
        if (player.specialCooldown > 0) {
            player.specialCooldown--;
        }
    }
    
    function drawStickman(player) {

        ctx.translate(player.x, player.y);
        if (!player.facingRight) {

            ctx.beginPath();

            ctx.fill();
            ctx.beginPath();
            ctx.ellipse(-20, 8, 6, 5, 0, 0, Math.PI * 2);
            ctx.fill();
        }
        
        ctx.strokeStyle = player.color;
        ctx.lineWidth = 3;
        ctx.lineCap = 'round';
        
        // Kopf
        ctx.beginPath();
        ctx.arc(0, -40, 10, 0, Math.PI * 2);
        ctx.stroke();
        
        // K√∂rper
        ctx.beginPath();
        ctx.moveTo(0, -30);
        ctx.lineTo(0, -10);
        ctx.stroke();
        
        // Arme
        ctx.beginPath();
        ctx.moveTo(0, -25);
        if (player.attacking) {
            ctx.lineTo(25, -20);
        } else {
            ctx.lineTo(15, -15);
        }
        ctx.stroke();
        
        ctx.beginPath();
        ctx.moveTo(0, -25);
        ctx.lineTo(-10, -20);
        ctx.stroke();
        
        // Beine
        ctx.beginPath();
        ctx.moveTo(0, -10);
        ctx.lineTo(-8, 0);

        ctx.moveTo(0, -10);

        ctx.lineWidth = 3;

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        // Boden
        ctx.fillStyle = '#0a2e4d';
        ctx.fillRect(0, groundY + 10, canvas.width, canvas.height - groundY - 10);
